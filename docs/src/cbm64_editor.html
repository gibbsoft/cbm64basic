 42class BasicLineValidator(Validator):
 43    """Validates BASIC program lines and returns errors with positions"""
 44
 45    def validate(self, document):
 46        text = document.text.strip()
 47
 48        # Skip empty lines
 49        if not text:
 50            return
 51
 52        # Check if it's a numbered line
 53        parts = text.split(None, 1)
 54        if not parts[0].isdigit():
 55            return  # Not a program line, let other validators handle it
 56
 57        line_num = int(parts[0])
 58        code = parts[1] if len(parts) > 1 else ""
 59
 60        # Line number only means delete - this is valid
 61        if not code:
 62            return
 63
 64        # Tokenize the line using the interpreter
 65        temp_interp = BasicInterpreter()
 66        full_line = f"{line_num} {code}"
 67        parsed_num, tokens = temp_interp.tokenize(full_line)
 68
 69        if not tokens:
 70            raise ValidationError(cursor_position=len(text), message="?SYNTAX ERROR")
 71
 72        command = tokens[0].upper()
 73
 74        # Check FOR syntax - must use = not in
 75        if command == "FOR":
 76            if len(tokens) < 5:
 77                raise ValidationError(
 78                    cursor_position=len(text), message="?SYNTAX ERROR"
 79                )
 80            # Check for "IN" (should be "=")
 81            # tokens format: ['FOR', 'varname', '=', 'start', 'TO', 'end']
 82            if tokens[2].upper() == "IN":
 83                # Find position of "IN" in the original text
 84                pos = code.find("in")
 85                if pos == -1:
 86                    pos = code.find("IN")
 87                if pos != -1:
 88                    raise ValidationError(
 89                        cursor_position=len(str(line_num)) + 2 + pos,
 90                        message="?USE = NOT IN",
 91                    )
 92
 93        # Check for invalid line number usage (commands that should be immediate)
 94        immediate_commands = [
 95            "LIST",
 96            "RUN",
 97            "NEW",
 98            "LOAD",
 99            "SAVE",
100            "CLR",
101            "CONT",
102            "DELETE",
103            "RENUM",
104            "AUTO",
105            "CLS",
106            "HELP",
107            "BYE",
108        ]
109        if command in immediate_commands:
110            raise ValidationError(
111                cursor_position=len(str(line_num)) + 1,
112                message=f"?SYNTAX ERROR IN {line_num}",
113            )

267def petscii_to_unicode(chars: str) -> str:
268    """Convert PETSCII character codes to Unicode glyphs."""
269    result = []
270    for char in chars:
271        code = ord(char)
272        if code in PETSCII_TO_UNICODE:
273            result.append(PETSCII_TO_UNICODE[code])
274        elif code < 128:
275            result.append(char)
276        else:
277            result.append(char)
278    return "".join(result)

342def format_basic_line(line_num: int, code: str) -> str:
343    """Format BASIC line: uppercase keywords, normalize whitespace"""
344
345    # Create temporary interpreter to access tokenizer
346    temp_interp = BasicInterpreter()
347
348    # Tokenize line
349    full_line = f"{line_num} {code}"
350    _, tokens = temp_interp.tokenize(full_line)
351
352    if not tokens:
353        return code
354
355    formatted_parts = []
356
357    for i, token in enumerate(tokens):
358        if i == 0 and token == str(line_num):
359            continue
360
361        if token.upper() in BASIC_KEYWORDS:
362            formatted_parts.append(token.upper())
363        else:
364            formatted_parts.append(token)
365
366    return " ".join(formatted_parts)

369def colorize_basic_line(line_num: int, code: str) -> FormattedText:
370    """Apply syntax highlighting to a BASIC line"""
371
372    # Create temporary interpreter to access tokenizer
373    temp_interp = BasicInterpreter()
374
375    # Tokenize line
376    full_line = f"{line_num} {code}"
377    _, tokens = temp_interp.tokenize(full_line)
378
379    result = []
380
381    # Add colored line number
382    result.append((COLORS["lineno"], str(line_num)))
383    result.append(("", " "))
384
385    for i, token in enumerate(tokens):
386        if i == 0 and token == str(line_num):
387            continue
388
389        token_upper = token.upper()
390
391        if token_upper == "REM":
392            # Everything after REM is a comment
393            result.append((COLORS["comment"], token_upper))
394        elif token_upper in BASIC_KEYWORDS:
395            result.append((COLORS["keyword"], token_upper))
396        elif token.startswith('"') and token.endswith('"'):
397            result.append((COLORS["string"], token))
398        elif token.replace(".", "").replace("-", "").isdigit() or (
399            token.replace(".", "").isdigit() and token[0] == "-"
400        ):
401            result.append((COLORS["number"], token))
402        elif token in ["+", "-", "*", "/", "=", "<", ">", "<=", ">=", "<>"]:
403            result.append((COLORS["operator"], token))
404        elif token in [";", ","]:
405            result.append(("", token))
406        else:
407            result.append((COLORS["variable"], token))
408
409        result.append(("", " "))
410
411    return FormattedText(result)

442    def parse_command(self, command: str):
443        """Parse and execute a single command - returns True if READY should be shown"""
444        command = command.strip()
445
446        if not command:
447            return False
448
449        parts = command.split()
450        if parts and parts[0].isdigit():
451            # Validate the program line first
452            validator = BasicLineValidator()
453            try:
454                from prompt_toolkit.document import Document
455
456                validator.validate(Document(text=command))
457            except ValidationError as e:
458                print(f"?{e.message}")
459                return False  # Don't show READY, line was rejected
460
461            self.handle_program_line(command)
462            return False  # Don't show READY after program lines
463        else:
464            self.handle_immediate_command(command)
465            return True  # Show READY after immediate commands

467    def handle_program_line(self, line: str):
468        """Handle a numbered program line - returns True if READY should be shown"""
469        parts = line.split(" ", 1)
470        if len(parts) >= 2:
471            line_num = int(parts[0])
472            code = parts[1].strip()
473
474            # Format line
475            formatted_code = format_basic_line(line_num, code)
476
477            # Check for immediate mode commands in program lines
478            command_part = formatted_code.split()[0] if formatted_code else ""
479            if command_part.upper() in [
480                "LIST",
481                "RUN",
482                "NEW",
483                "LOAD",
484                "SAVE",
485                "CLR",
486                "CONT",
487                "DELETE",
488                "RENUM",
489                "AUTO",
490                "CLS",
491            ]:
492                # Move cursor up one line, clear it, and show error
493                if not self._suppress_cursor_magic:
494                    sys.stdout.write("\033[F\033[2K")
495                    sys.stdout.flush()
496                if sys.stdout.isatty():
497                    print_formatted_text(
498                        FormattedText([("ansired", f"?SYNTAX ERROR IN {line_num}")])
499                    )
500                else:
501                    print(f"?SYNTAX ERROR IN {line_num}")
502            else:
503                # Store formatted version
504                self.program_lines[line_num] = formatted_code
505                # Move cursor up one line, clear it, and print formatted version
506                if not self._suppress_cursor_magic:
507                    sys.stdout.write("\033[F\033[2K")
508                    sys.stdout.flush()
509                if sys.stdout.isatty():
510                    print_formatted_text(colorize_basic_line(line_num, formatted_code))
511                else:
512                    print(f"{line_num} {formatted_code}")
513        else:
514            # Just a line number - delete line
515            line_num = int(parts[0])
516            if line_num in self.program_lines:
517                del self.program_lines[line_num]
518                print(f"{line_num} DELETED")

520    def handle_immediate_command(self, command: str):
521        """Handle immediate mode commands"""
522        cmd_upper = command.upper()
523
524        if cmd_upper == "LIST":
525            self.list_program()
526        elif cmd_upper == "RUN":
527            self.run_program()
528        elif cmd_upper == "NEW":
529            self.new_program()
530        elif cmd_upper == "LOAD":
531            self.load_program()
532        elif cmd_upper == "SAVE":
533            self.save_program()
534        elif cmd_upper == "CLR":
535            self.clear_variables()
536        elif cmd_upper == "HELP":
537            self.show_help()

539            print("GOODBYE!")

542            self.handle_edit_command(command)

562        """RUN command - execute the current program"""
563        if not self.program_lines:

578            self.interpreter.prompt_func = None

729    def handle_multiline_input(self, text: str) -> bool:
730        """Handle multi-line input (e.g., pasted code).
731
732        Returns True if READY should be shown.
733        """
734        # Suppress cursor magic during multi-line paste to avoid escape sequence interference
735        old_suppress = self._suppress_cursor_magic
736        self._suppress_cursor_magic = True
737
738        lines = text.strip().split("\n")
739        show_ready = False
740
741        for line in lines:
742            line = line.strip()
743            if line:
